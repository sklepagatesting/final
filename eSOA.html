<script>
let textTimeout;
let currentFocusedIndex = 0;
const allRows = document.querySelectorAll('.item-row');
const highlighter = document.getElementById('highlighter');
const container = document.getElementById('mainContainer');
const magnetBtn = document.getElementById('magnetBtn');
const modalCopyBtn = document.getElementById('modalCopyBtn');
const srModal = document.getElementById('srModal');
const modalBox = document.getElementById('modalBox');

window.onload = () => {
setTimeout(() => updateFocus(0), 100);
allRows.forEach((row, idx) => {
row.addEventListener('click', () => executeCopy(row, row.getAttribute('data-val'), idx));
});
lucide.createIcons();
};

function toggleModal(show) {
srModal.style.display = show ? 'flex' : 'none';
if(show) {
document.getElementById('hospCharges').focus();
} else {
resetCalculator();
}
}

// Auto-close when clicking background overlay
srModal.addEventListener('click', (e) => {
if (e.target === srModal) toggleModal(false);
});

function resetCalculator() {
document.getElementById('hospCharges').value = "";
document.getElementById('caseRate').value = "";
document.getElementById('finalDiscount').innerText = "0.00";
}

function calcSr() {
const charges = parseFloat(document.getElementById('hospCharges').value) || 0;
const rate = parseFloat(document.getElementById('caseRate').value) || 0;
const discount = charges - ((charges * 0.20) + rate);
document.getElementById('finalDiscount').innerText = discount.toFixed(2);
}

function copyModalValue() {
const val = document.getElementById('finalDiscount').innerText;
navigator.clipboard.writeText(val).then(() => {
modalCopyBtn.innerText = "COPIED!";
modalCopyBtn.classList.add('success-state');
setTimeout(() => {
modalCopyBtn.innerText = "COPY RESULT";
modalCopyBtn.classList.remove('success-state');
toggleModal(false); // Auto-close after copy
}, 600);
});
}

/* 🐾 IDLE SYSTEM */
let idleTimer, nudgeTimer;
function startIdle() {
magnetBtn.classList.add('idle');
nudgeTimer = setInterval(() => {
magnetBtn.classList.add('nudge');
setTimeout(() => magnetBtn.classList.remove('nudge'), 600);
}, 4000);
}
function stopIdle() {
magnetBtn.classList.remove('idle', 'nudge');
clearInterval(nudgeTimer);
}
function resetIdle() {
stopIdle();
clearTimeout(idleTimer);
idleTimer = setTimeout(startIdle, 1500);
}
['mousemove','click','scroll'].forEach(e => document.addEventListener(e, resetIdle));
setTimeout(startIdle, 1200);

/* MAGNET LOGIC */
document.addEventListener('mousemove', (e) => {
const threshold = window.innerHeight * 0.75;
if (e.clientY > threshold) {
const centerX = window.innerWidth / 2;
const centerY = window.innerHeight - (window.innerHeight * 0.125);
const pullX = (e.clientX - centerX) * 0.3;
const pullY = (e.clientY - centerY) * 0.4;
magnetBtn.style.transform = `translate(${pullX}px, ${pullY}px)`;
} else {
resetMagnet();
}
});

function resetMagnet() { magnetBtn.style.transform = `translate(0, 0)`; }

function updateFocus(index) {
currentFocusedIndex = index;
const targetRow = allRows[index];
const rect = targetRow.getBoundingClientRect();
const containerRect = container.getBoundingClientRect();
highlighter.style.opacity = "1";
highlighter.style.top = (rect.top - containerRect.top) + "px";
highlighter.style.height = rect.height + "px";
targetRow.scrollIntoView({ behavior: 'smooth', block: 'center' });
}

function executeCopy(row, text, index) {
updateFocus(index);
navigator.clipboard.writeText(text).then(() => {
document.querySelectorAll('.text-content').forEach(el => el.classList.remove('text-active'));
clearTimeout(textTimeout);
const textElement = row.querySelector('.text-content');
textElement.classList.add('text-active');
magnetBtn.innerText = "COPIED!";
magnetBtn.classList.add('success-state');
document.body.classList.add('copy-flash');
setTimeout(() => {
magnetBtn.innerText = "COPY";
magnetBtn.classList.remove('success-state');
document.body.classList.remove('copy-flash');
}, 600);
setTimeout(() => nextFocus(), 400);
textTimeout = setTimeout(() => textElement.classList.remove('text-active'), 8000);
});
}

function copyFocusedItem() {
const activeRow = allRows[currentFocusedIndex];
executeCopy(activeRow, activeRow.getAttribute('data-val'), currentFocusedIndex);
}

function nextFocus() {
currentFocusedIndex = (currentFocusedIndex + 1) % allRows.length;
updateFocus(currentFocusedIndex);
}

function resetFocus() {
const fbsElement = document.getElementById('reset-target');
const fbsIndex = Array.from(allRows).indexOf(fbsElement);
updateFocus(fbsIndex);
}
